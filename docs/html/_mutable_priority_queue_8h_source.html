112template <class T>
113void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
114 auto x = H[i];
115 while (i > 1 && *x < *H[parent(i)]) {
116 set(i, H[parent(i)]);
117 i = parent(i);
118 }
119 set(i, x);
120}

122template <class T>
123void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
124 auto x = H[i];
125 while (true) {
126 unsigned k = leftChild(i);
127 if (k >= H.size())
128 break;
129 if (k+1 < H.size() && *H[k+1] < *H[k])
130 ++k; // right child of i
131 if ( ! (*H[k] < *x) )
132 break;
133 set(i, H[k]);
134 i = k;
135 }
136 set(i, x);
137}
