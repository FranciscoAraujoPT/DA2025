352/*
353 * Auxiliary function to remove an outgoing edge (with a given destination (d))
354 * from a vertex (this).
355 * Returns true if successful, and false if such edge does not exist.
356 */
357template <class T>
358bool Vertex<T>::removeEdge(T in) {
359 bool removedEdge = false;
360 auto it = adj.begin();
361 while (it != adj.end()) {
362 Edge<T> *edge = *it;
363 Vertex<T> *dest = edge->getDest();
364 if (dest->getInfo() == in) {
365 it = adj.erase(it);
366 deleteEdge(edge);
367 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
368 }
369 else {
370 it++;
371 }
372 }
373 return removedEdge;
374}

377 * Auxiliary function to remove an outgoing edge of a vertex.
378 */
379template <class T>
380void Vertex<T>::removeOutgoingEdges() {
381 auto it = adj.begin();
382 while (it != adj.end()) {
383 Edge<T> *edge = *it;
384 it = adj.erase(it);
385 deleteEdge(edge);
386 }
387}

495void Vertex<T>::deleteEdge(Edge<T> *edge) {
496 Vertex<T> *dest = edge->getDest();
497 // Remove the corresponding edge from the incoming list
498 auto it = dest->incoming.begin();
499 while (it != dest->incoming.end()) {
500 if ((*it)->getOrig()->getInfo() == info) {

502 }
503 else {
504 it++;
505 }
506 }
507 delete edge;

595/*
596 * Removes a vertex with a given content (in) from a graph (this), and
597 * all outgoing and incoming edges.
598 * Returns true if successful, and false if such vertex does not exist.
599 */
600template <class T>
601bool Graph<T>::removeVertex(const T &in) {
602 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
603 if ((*it)->getInfo() == in) {
604 auto v = *it;
605 v->removeOutgoingEdges();
606 for (auto u : vertexSet) {
607 u->removeEdge(v->getInfo());
608 }
609 vertexSet.erase(it);
610 delete v;
611 return true;
612 }
613 }
614 return false;
615}

646template <class T>
647bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double walking, double driving) {
648 auto v1 = findVertex(sourc);
649 auto v2 = findVertex(dest);
650 if (v1 == nullptr || v2 == nullptr)
651 return false;
652 auto e1 = v1->addEdge(v2, walking, driving);
653 auto e2 = v2->addEdge(v1, walking, driving);
654 e1->setReverse(e2);
655 e2->setReverse(e1);
656 return true;
